689  public function Exec(bool $responseProcessing = true): void {
690  if (!extension_loaded('curl')) { return; }
691  $active = null;
692  do {
693  $mrc = curl_multi_exec($this->curl_multi, $active);
694  } while ($mrc == CURLM_CALL_MULTI_PERFORM || $active);
695 
696  if($mrc == CURLM_OK && !$active) {
697  unset($mrc, $active);
698  foreach ($this->curl_handles as $call => $hashs) {
699  foreach ($hashs as $hash => $curl) {
700  $this->server_data['json'][$call][$hash] = curl_multi_getcontent($curl);
701  $this->server_data['http_status_code'][$call][$hash]['code'] = curl_getinfo($curl, CURLINFO_HTTP_CODE);
702  curl_multi_remove_handle($this->curl_multi, $curl); //remove
703  if ($this->server_data['http_status_code'][$call][$hash]['code'] == 200) {
704  if ($this->curl_config[$call][$hash]['gzip']) {
705  $sections = explode("\x0d\x0a\x0d\x0a", $this->server_data['json'][$call][$hash], 2);
706  while (!strncmp($sections[1], 'HTTP/', 5)) {
707  $sections = explode("\x0d\x0a\x0d\x0a", $sections[1], 2);
708  }
709 
710  if (count($sections) >= 2) {
711  if (preg_match('/^Content-Encoding: gzip/mi', $sections[0])) {
712  $this->server_data['json'][$call][$hash] = $sections[1];
713  }
714  }
715  }
716 
717  if ($responseProcessing) {
718  $this->responseProcessing($call,$hash); //processing data
719  }
720 
721  //execution time
722  $this->server_data['http_status_code'][$call][$hash]['execution_time'] =
723  (microtime(true) - $this->curl_config[$call][$hash]['time']);
724  }
725 
726  curl_close($this->curl_handles[$call][$hash]);
727  unset($this->curl_handles[$call][$hash], $this->curl_config[$call][$hash]);
728  }
729  }
730 
731  curl_multi_close($this->curl_multi);
732  $this->curl_multi = curl_multi_init();
733  } else {
